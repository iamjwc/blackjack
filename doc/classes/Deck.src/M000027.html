  def add_deck!
    Card::SUITS.each do |suit|
      Card::VALUES.each do |value|
        # Inserts each card into a random position between 0 and the current
        # number of cards already in @cards.
        @cards.insert( rand(@cards.size), Card.new(suit, value) )
      end
    end

    self
  end

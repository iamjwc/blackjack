  def play
    all_players.each do |p|
      while p.active?
        case p.take_turn( @dealer.hand[1].dup )
          when :hit:         p.hit( @deck.take_card )
          when :double_down: p.double_down( @deck.take_card )
          when :stand:       p.stand
        end
      end
    end

    # If the dealer busts, the remaining players win
  end

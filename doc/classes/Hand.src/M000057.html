  def split
    if split_allowed?
      hands = [Hand.new(@player),Hand.new(@player)]

      b = @bet
      clear_bet!
      
      2.times do |i|
        hands[i] << @hand[i]
        hands[i].bet = b
        hands[i].from_split = true
      end

      @invalid = true

      hands
    else
      raise 'Split not allowed.'
    end
  end

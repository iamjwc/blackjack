  def test_face_card?
    Card::SUITS.each do |suit|
      Card::VALUES.each do |value|
        if [:king,:queen,:jack].include? value
          # Asserts that these cards are face cards
          assert Card.new(suit, value).face_card?
        else
          # Asserts that these cards are NOT face cards
          assert !Card.new(suit, value).face_card?
        end
      end
    end
  end

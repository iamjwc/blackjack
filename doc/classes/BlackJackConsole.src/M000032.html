  def run
    Interface::header

    menu_selection = nil
    until :quit == menu_selection
      Interface::newline
      Interface::seperator
      count = Interface::input("How many players? {1-4}: ") do |o|
        (1..4) === o.to_i
      end

      Interface::newline
      Interface::seperator
      count.to_i.times do |i|
        name = Interface::input("Player #{i+1}'s name: ") do |o|
          !o.empty?
        end

        @b.join( HumanPlayer.new(name) )
      end

      Interface::newline
      @b.play_game

      menu_selection = Interface::main_menu
    end
  end
